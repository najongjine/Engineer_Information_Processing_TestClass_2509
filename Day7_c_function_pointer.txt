#include <stdio.h>

void swap(int _a, int _b){
    int temp=_a;
    _a=_b;
    _b=_a;
}
int main() {
    int a=1;
    int b=2;
    swap(a,b);
    printf("a:%d, b:%d \n",a,b)

    return 0;
}


main에 있는 a 와 b가 서로 값을 교환함
a의 주소, b의 주소를 주니깐(본체를 주니깐) 값 교환이 일어남
#include <stdio.h>

void swap(int *_a, int *_b){
    int temp=*_a;
    *_a=*_b;
    *_b=temp;
}
int main() {
    int a=1;
    int b=2;
    swap(&a,&b);
    printf("a:%d, b:%d \n",a,b);

    return 0;
}
