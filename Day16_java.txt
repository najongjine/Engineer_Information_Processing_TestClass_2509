public class Test14 {
	public static void main(String[] args) {
		new Examp().aaa(10);
		new Examp().aa(new int[]{1,2,3,4,5});
		new Examp().a(new MyExam3());//1
	}
}
class Examp{
	void aaa(int a) {		System.out.println(a);	}
	void aa(int[] a) {		System.out.println(Arrays.toString(a));	}
	void a(ExamIntf e) {//2
		e.getVersion();//3
	}
}
interface ExamIntf{	public void getVersion();}
class MyExam implements ExamIntf{
	private String version="v1";
	public void getVersion() {		System.out.println(this.version);	}
}
class MyExam2 implements ExamIntf{
	private String version="v2";
	void mygetVersion() {		System.out.println(this.version);	}
	public void getVersion() {		mygetVersion();	}
}
class MyExam3 implements ExamIntf{
	public void getVersion() {
		aaa();
	}
	void aaa() {
		bbb();
	}
	void bbb() {
		System.out.println("V3");
	}	
}
