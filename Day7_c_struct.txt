#include <stdio.h>

// myChild 라는 구조체(폴더랑 비슷함. java 의 class 하위호환)
struct myChild {
    int myNum;
    char myLetter;
};

// myStructure 라는 구조체(폴더랑 비슷함. java 의 class 하위호환)
// 현 구조는 폴더 안에 폴더가 있는 구조
struct myStructure {
    int myNum;
    char myLetter;
    struct myChild my_child;  // myChild 라는 타입에 my_child 라는 변수
};


int main() {
    struct myStructure s1;
    s1.myNum=0;
    s1.myLetter='a';
    s1.my_child.myNum=1;
    printf("s1.myNum: %d \n",s1.myNum);

    return 0;
}


/* s1 이랑, s2 랑 완전히 별개에요. 서로 다른 메모리 공간을 가지고 있어요 */
struct myStructure s1;
struct myStructure s2;
s1.myNum=0;
s2.myNum=1;

printf("s1.myNum: %d \n",s1.myNum); //0
printf("s2.myNum: %d \n",s2.myNum); //1






#include <stdio.h>

struct Car {
  char brand[50];
  int year;
};

int main() {
    struct Car car = {"Toyota", 2020};

    struct Car *ptr = &car;
    
    /* 그냥 평범한 struct 변수면 . 찍어서 속성에 접근하면 됨 */
    car.year;
    
    /* 포인터 struct 이면 -> 써서 속성에 접근하면 됨 */
    printf("Brand: %s\n", ptr->brand);
    printf("Year: %d\n", ptr->year);

    return 0;
}
