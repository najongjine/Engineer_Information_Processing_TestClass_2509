DCL -
계정, 권한
CREATE DATABASE test1; -- 데이터베이스 만들기(건물 세우기)
 -- CREATE DATABASE 이거는 표준 SQL 외워야함

CREATE USER 'testuser'@'localhost' IDENTIFIED BY 'aaaa';
-- CREATE USER 까지는 공통.
-- 유저이름은 앞쪽, 비밀번호는 뒤쪽


-- 특정 DB에 대한 모든 권한 부여
GRANT ALL PRIVILEGES ON test1.* TO 'testuser'@'localhost';
-- GRANT ALL PRIVILEGES ON dcl_test.* TO 'testuser'

-- SELECT 권한만 부여
GRANT SELECT ON test1.* TO 'testuser'@'localhost';


GRANT ALL PRIVILEGES ON mydb.mytable TO 'user_a'@'localhost' WITH GRANT OPTION;
-- WITH GRANT OPTION; 권한을 주면서, 권한 주는 기능도 함께 주기

-- 유저의 권한보기
SHOW GRANTS FOR 'testuser'@'localhost';
-- SHOW GRTANTS 까지는 공통


-- 유저의 권한 뺏기
REVOKE SELECT ON test1.* FROM 'testuser'@'localhost';
-- REVOKE

-- 사용자 삭제
DROP USER 'testuser'@'localhost';
-- DROP USER


-- DB 삭제
DROP DATABASE test1;
-- DROP DATABASE



DDL -
테이블 생성, 변경, 삭제

DML -
데이터 가져오기, 데이터 수정+추가 하기, 데이터 삭제하기
