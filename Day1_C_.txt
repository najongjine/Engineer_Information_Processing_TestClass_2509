주석은 컴퓨터가 실행을 안해요
*/
/* 패키지 -
* 이미 만들어진 코드 뭉탱이를 가져오기
c언어 쓸때 #include <stdio.h> 이건 무조건 있어야함

* 외부에서 만들어진 컴포넌트나 모듈의 모음
*/
#include <stdio.h>

/* C 언어하고 자바는 무조건 main 함수가 있어야
해요*/
int main() {
    /* 출력 함수
    * 이거 해줘야 보여짐
    시험 문제에서는 print() 가 정답 쓰라는 뜻
    \n 은 다음줄 개행(줄바꾸기)*/
    printf("Try programiz.pro \n");
    
    /*
    상수. 바뀌지 않는 값
    */
    3;
    
    /*
    변수 - 값이 바뀜
    세미콜론(;) - 컴퓨터한테 이게 한줄이야. 실행
    해야되 하고 알려 주는 키워드
    */
    int a;
    //printf("a: %d\n",a);
    a=2;
    
    printf("a: %d\n",a);
    
    // 과제 : b 라는 변수를 int 형으로 선언해 주세요
    int b;
    /*
    b 라는 int 형 변수를 선언 했는데,
    또 b 라는 int형 변수를 선언하면
    컴퓨터 입장에선, 첫번째 b인지, 두번째 b인지
    알수가 없기 때문에 허용 자체를 안해요
    
    자료형 변수이름 : 이것을
    변수를 선언한다 라고 합니다
    */
    //int b;
    b=22;
    printf("b의 값: %d \n",b);
     
     /*
     사람이 메모리 주소를 직접 쓰면서 접근하는건 
     불가능 합니다.
     그래서 변수라는 시스템을 만든 거에요
     */
    
    int c;
    printf("초기 c의 값: %d \n",c);
    
     /*
    c 라는 변수의 메모리 공간에
    b 라는 변수의 메모레이 있는 값을
    c 에 저장해 주세요
    * 대입연산
    */
    c = b;
    printf("c의 값: %d \n",c);
    
    /*
    c 언어의 자료형
    int - 숫자
    float - 소수점
    char - 문자
    */
    // c,java 에선 소수점만 담음
    float f1=3.14;
    printf(" float size: %d \n",sizeof(f1));
    printf(" f1: %f \n",f1);
    // c,java 언어에서 int 는 정수만 담음
    int a2=1/2;
    printf(" a2: %d \n",a2);

    return 0;
}
