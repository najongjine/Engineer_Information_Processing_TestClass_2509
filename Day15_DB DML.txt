-- Departments 테이블 생성
CREATE TABLE Departments (
    DeptID INT PRIMARY KEY,
    DeptName VARCHAR(50) NOT NULL,
    Location VARCHAR(50)
);

---

-- Employees 테이블 생성
CREATE TABLE Employees (
    EmpID INT PRIMARY KEY,
    EmpName VARCHAR(100) NOT NULL,
    DeptID INT,
    HireDate DATE,
    Salary DECIMAL(10, 2), -- 급여는 소수점 두 자리까지 가능하도록 DECIMAL 사용
    JobTitle VARCHAR(100),
    FOREIGN KEY (DeptID) REFERENCES Departments(DeptID) -- DeptID는 Departments 테이블을 참조하는 외래 키
);

---

-- Projects 테이블 생성
CREATE TABLE Projects (
    ProjID INT PRIMARY KEY,
    ProjName VARCHAR(100) NOT NULL,
    DeptID INT, -- 프로젝트를 관리하는 부서 ID로 가정
    StartDate DATE,
    EndDate DATE,
    Budget DECIMAL(12, 2), -- 예산은 좀 더 큰 값도 수용할 수 있도록 DECIMAL(12, 2) 사용
    FOREIGN KEY (DeptID) REFERENCES Departments(DeptID) -- DeptID는 Departments 테이블을 참조하는 외래 키
);
